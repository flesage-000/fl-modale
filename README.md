## Install

```bash
npm install --save fl-modale
```

## Usage

```jsx
import React, { Component } from 'react'

import MyComponent from 'fl-modale'
import 'fl-modale/dist/index.css'

class Example extends Component {
  render() {
    return <MyComponent />
  }
}
```
